24 void
25 depthmap_cleanup_grow (FloatImage::ConstPtr dm, FloatImage::Ptr ret,
26  std::vector<bool>& visited,
27  std::size_t x, std::size_t y, std::size_t thres)
28 {
29  typedef std::list<std::size_t> PixelQueue;
30  typedef std::set<std::size_t> PixelBag;
31 
32  std::size_t w = dm->width();
33  std::size_t h = dm->height();
34  std::size_t idx = y * w + x;
35  std::size_t max_idx = w * h;
36  std::size_t const MAX_ST = (std::size_t)-1;
37 
38  PixelQueue queue;
39  PixelBag collected;
40 
41  queue.push_front(idx);
42  collected.insert(idx);
43 
44  /* Process queue of pixel until either thres is reached or no more
45  * pixels are available for growing. If thres is reached, mark as
46  * visited and leave pixels. If no more pixels are available, an
47  * isolated small island has been found and pixels are set to zero.
48  */
49  while (!queue.empty())
50  {
51  /* Take queue element and process... */
52  std::size_t cur = queue.back();
53  queue.pop_back();
54 
55  /* Define 4-neighborhood for region growing. Y-1, Y+1, X-1, X+1. */
56  std::size_t n[4];
57  n[0] = (cur < w ? MAX_ST : cur - w);
58  n[1] = (cur < max_idx - w ? cur + w : MAX_ST);
59  n[2] = (cur % w > 0 ? cur - 1 : MAX_ST);
60  n[3] = (cur % w < w - 1 ? cur + 1 : MAX_ST);
61 
62  /* Invalidate unreconstructed neighbors. */
63  for (int i = 0; i < 4; ++i)
64  if (n[i] != MAX_ST && dm->at(n[i], 0) == 0.0f)
65  n[i] = MAX_ST;
66 
67  /* Add uncollected pixels to queue. */
68  for (int i = 0; i < 4; ++i)
69  if (n[i] != MAX_ST && collected.find(n[i]) == collected.end())
70  {
71  queue.push_front(n[i]);
72  collected.insert(n[i]);
73  }
74  }
75 
76  /* Mark collected region as visited and set depth values to zero
77  * if amount of collected pixels is less than threshold.
78  */
79  for (PixelBag::iterator i = collected.begin(); i != collected.end(); i++)
80  {
81  visited[*i] = true;
82  if (collected.size() < thres)
83  ret->at(*i, 0) = 0.0f;
84  }
85 }

188 bool
189 dm_is_depthdisc (float* widths, float* depths, float dd_factor, int i1, int i2)
190 {
191  /* Find index that corresponds to smaller depth. */
192  int i_min = i1;
193  int i_max = i2;
194  if (depths[i2] < depths[i1])
195  std::swap(i_min, i_max);
196 
197  /* Check if indices are a diagonal. */
198  if (i1 + i2 == 3)
199  dd_factor *= MATH_SQRT2;
200 
201  /* Check for depth discontinuity. */
202  if (depths[i_max] - depths[i_min] > widths[i_min] * dd_factor)
203  return true;
204 
205  return false;
206 }

210 TriangleMesh::Ptr
211 depthmap_triangulate (FloatImage::ConstPtr dm, math::Matrix3f const& invproj,
212  float dd_factor, mve::Image<unsigned int>* vids)
213 {
214  if (dm == nullptr)
215  throw std::invalid_argument("Null depthmap given");
216 
217  int const width = dm->width();
218  int const height = dm->height();
219 
220  /* Prepare triangle mesh. */
221  TriangleMesh::Ptr mesh(TriangleMesh::create());
222 
223  /* Generate image that maps image pixels to vertex IDs. */
224  mve::Image<unsigned int> vidx(width, height, 1);
225  vidx.fill(MATH_MAX_UINT);
226 
227  /* Iterate over 2x2-blocks in the depthmap and create triangles. */
228  int i = 0;
229  for (int y = 0; y < height - 1; ++y, ++i)
230  {
231  for (int x = 0; x < width - 1; ++x, ++i)
232  {
233  /* Cache the four depth values. */
234  float depths[4] = { dm->at(i, 0), dm->at(i + 1, 0),
235  dm->at(i + width, 0), dm->at(i + width + 1, 0) };
236 
237  /* Create a mask representation of the available depth values. */
238  int mask = 0;
239  int pixels = 0;
240  for (int j = 0; j < 4; ++j)
241  if (depths[j] > 0.0f)
242  {
243  mask |= 1 << j;
244  pixels += 1;
245  }
246 
247  /* At least three valid depth values are required. */
248  if (pixels < 3)
249  continue;
250 
251  /* Possible triangles, vertex indices relative to 2x2 block. */
252  int tris[4][3] = {
253  { 0, 2, 1 }, { 0, 3, 1 }, { 0, 2, 3 }, { 1, 2, 3 }
254  };
255 
256  /* Decide which triangles to issue. */
257  int tri[2] = { 0, 0 };
258 
259  switch (mask)
260  {
261  case 7: tri[0] = 1; break;
262  case 11: tri[0] = 2; break;
263  case 13: tri[0] = 3; break;
264  case 14: tri[0] = 4; break;
265  case 15:
266  {
267  /* Choose the triangulation with smaller diagonal. */
268  float ddiff1 = std::abs(depths[0] - depths[3]);
269  float ddiff2 = std::abs(depths[1] - depths[2]);
270  if (ddiff1 < ddiff2)
271  { tri[0] = 2; tri[1] = 3; }
272  else
273  { tri[0] = 1; tri[1] = 4; }
274  break;
275  }
276  default: continue;
277  }
278 
279  /* Omit depth discontinuity detection if dd_factor is zero. */
280  if (dd_factor > 0.0f)
281  {
282  /* Cache pixel footprints. */
283  float widths[4];
284  for (int j = 0; j < 4; ++j)
285  {
286  if (depths[j] == 0.0f)
287  continue;
288  widths[j] = pixel_footprint(x + (j % 2), y + (j / 2),
289  depths[j], invproj);// w, h, focal_len);
290  }
291 
292  /* Check for depth discontinuities. */
293  for (int j = 0; j < 2 && tri[j] != 0; ++j)
294  {
295  int* tv = tris[tri[j] - 1];
296  #define DM_DD_ARGS widths, depths, dd_factor
297  if (dm_is_depthdisc(DM_DD_ARGS, tv[0], tv[1])) tri[j] = 0;
298  if (dm_is_depthdisc(DM_DD_ARGS, tv[1], tv[2])) tri[j] = 0;
299  if (dm_is_depthdisc(DM_DD_ARGS, tv[2], tv[0])) tri[j] = 0;
300  }
301  }
302 
303  /* Build triangles. */
304  for (int j = 0; j < 2; ++j)
305  {
306  if (tri[j] == 0) continue;
307  #define DM_MAKE_TRI_ARGS mesh.get(), vidx, dm.get(), invproj, i
308  dm_make_triangle(DM_MAKE_TRI_ARGS, tris[tri[j] - 1]);
309  }
310  }
311  }
312 
313  /* Provide the vertex ID mapping if requested. */
314  if (vids != nullptr)
315  std::swap(vidx, *vids);
316 
317  return mesh;
318 }

404 bool
405 dm_is_depth_disc (math::Vec3f const& v1,
406  math::Vec3f const& v2, math::Vec3f const& v3)
407 {
408  float const angle_threshold = MATH_DEG2RAD(15.0f);
409  /* Depth discontinuity detection based on minimal angle in triangle. */
410  math::Vec3f e[3] = { (v2 - v1).normalized(),
411  (v3 - v2).normalized(), (v1 - v3).normalized() };
412  float min_angle = angle_threshold;
413  for (int i = 0; i < 3; ++i)
414  min_angle = std::min(min_angle, std::acos(e[i].dot(-e[(i + 1) % 3])));
415  //std::cout << "Min angle is " << MATH_RAD2DEG(min_angle) << ", "
416  // << (min_angle < ANGLE_THRES) << std::endl;
417  return min_angle < angle_threshold;
418 }

420 void
421 rangegrid_triangulate (Image<unsigned int> const& grid, TriangleMesh::Ptr mesh)
422 {
423  if (mesh == nullptr)
424  throw std::invalid_argument("Null mesh given");
425 
426  int w = grid.width();
427  int h = grid.height();
428 
429  TriangleMesh::VertexList const& verts(mesh->get_vertices());
430  TriangleMesh::FaceList& faces(mesh->get_faces());
431 
432  for (int y = 0; y < h - 1; ++y)
433  for (int x = 0; x < w - 1; ++x)
434  {
435  int i = y * w + x;
436 
437  unsigned int vid[4] = { grid[i], grid[i + 1],
438  grid[i + w], grid[i + w + 1] };
439 
440  /* Create a mask representation of the available indices. */
441  int mask = 0;
442  int indices = 0;
443  for (int j = 0; j < 4; ++j)
444  if (vid[j] != static_cast<unsigned int>(-1))
445  {
446  mask |= 1 << j;
447  indices += 1;
448  }
449 
450  /* At least three valid depth values are required. */
451  if (indices < 3)
452  continue;
453 
454  /* Decide which triangles to issue.
455  * Triangle 1: (0,2,1), 2: (0,3,1), 3: (0,2,3), 4: (1,2,3) */
456  int tri[2] = { 0, 0 };
457 
458  switch (mask)
459  {
460  case 7: tri[0] = 1; break;
461  case 11: tri[0] = 2; break;
462  case 13: tri[0] = 3; break;
463  case 14: tri[0] = 4; break;
464  case 15:
465  {
466  /* Choose the triangulation with smaller diagonal. */
467  if ((verts[vid[0]] - verts[vid[3]]).square_norm() <
468  (verts[vid[1]] - verts[vid[2]]).square_norm())
469  { tri[0] = 2; tri[1] = 3; }
470  else
471  { tri[0] = 1; tri[1] = 4; }
472  break;
473  }
474  default: continue;
475  }
476 
477  /* Check for depth discontinuities and issue triangles. */
478  for (int j = 0; j < 2; ++j)
479  {
480  switch (tri[j])
481  {
482 #define VERT(id) verts[vid[id]]
483 #define DEPTHDISC(a,b,c) dm_is_depth_disc(VERT(a), VERT(b), VERT(c))
484 #define ADDTRI(a,c,b) faces.push_back(vid[a]); faces.push_back(vid[b]); faces.push_back(vid[c])
485  case 1: if (!DEPTHDISC(0,2,1)) { ADDTRI(0,2,1); } break;
486  case 2: if (!DEPTHDISC(0,3,1)) { ADDTRI(0,3,1); } break;
487  case 3: if (!DEPTHDISC(0,2,3)) { ADDTRI(0,2,3); } break;
488  case 4: if (!DEPTHDISC(1,2,3)) { ADDTRI(1,2,3); } break;
489  default: break;
490  }
491  }
492  }
493 }

497 void
498 depthmap_mesh_confidences (TriangleMesh::Ptr mesh, int iterations)
499 {
500  if (mesh == nullptr)
501  throw std::invalid_argument("Null mesh given");
502 
503  if (iterations < 0)
504  throw std::invalid_argument("Invalid amount of iterations");
505 
506  if (iterations == 0)
507  return;
508 
509  TriangleMesh::ConfidenceList& confs(mesh->get_vertex_confidences());
510  TriangleMesh::VertexList const& verts(mesh->get_vertices());
511  confs.clear();
512  confs.resize(verts.size(), 1.0f);
513 
514  /* Find boundary vertices and remember them. */
515  std::vector<std::size_t> vidx;
516  MeshInfo mesh_info(mesh);
517 
518  for (std::size_t i = 0; i < mesh_info.size(); ++i)
519  {
520  if (mesh_info[i].vclass == MeshInfo::VERTEX_CLASS_BORDER)
521  vidx.push_back(i);
522  }
523 
524  /* Iteratively expand the current region and update confidences. */
525  for (int current = 0; current < iterations; ++current)
526  {
527  /* Calculate confidence for that iteration. */
528  float conf = (float)current / (float)iterations;
529  //conf = math::algo::fastpow(conf, 3);
530 
531  /* Assign current confidence to all vertices. */
532  for (std::size_t i = 0; i < vidx.size(); ++i)
533  confs[vidx[i]] = conf;
534 
535  /* Replace vertex list with adjacent vertices. */
536  std::vector<std::size_t> cvidx;
537  std::swap(vidx, cvidx);
538  for (std::size_t i = 0; i < cvidx.size(); ++i)
539  {
540  MeshInfo::VertexInfo info = mesh_info[cvidx[i]];
541  for (std::size_t j = 0; j < info.verts.size(); ++j)
542  if (confs[info.verts[j]] == 1.0f)
543  vidx.push_back(info.verts[j]);
544  }
545  }
546 }
