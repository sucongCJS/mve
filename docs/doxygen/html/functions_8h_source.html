233 template <typename T>
234 T
235 fastpow (T const& base, unsigned int exp)
236 {
237  T ret = (exp == 0 ? T(1) : base);
238  unsigned int const exp2 = exp / 2;
239  unsigned int i = 1;
240  for (; i <= exp2; i = i * 2)
241  ret = ret * ret;
242  for (; i < exp; ++i)
243  ret = ret * base;
244  return ret;
245 }
