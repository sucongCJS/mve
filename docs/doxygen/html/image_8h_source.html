232 template <typename T>
233 /*inline*/ void
234 Image<T>::add_channels (int num_channels, T const& value)
235 {
236  if (num_channels <= 0 || !this->valid())
237  return;
238 
239  std::vector<T> tmp(this->w * this->h * (this->c + num_channels));
240  typename std::vector<T>::iterator dest_ptr = tmp.end();
241  typename std::vector<T>::const_iterator src_ptr = this->data.end();
242  const int pixels = this->get_pixel_amount();
243  for (int p = 0; p < pixels; ++p)
244  {
245  for (int i = 0; i < num_channels; ++i)
246  *(--dest_ptr) = value;
247  for (int i = 0; i < this->c; ++i)
248  *(--dest_ptr) = *(--src_ptr);
249  }
250 
251  this->c += num_channels;
252  std::swap(this->data, tmp);
253 }

255 template <typename T>
256 void
257 Image<T>::swap_channels (int c1, int c2)
258 {
259  if (!this->valid() || c1 == c2
260  || c1 >= this->channels() || c2 >= this->channels())
261  return;
262 
263  T* iter1 = &this->data[0] + c1;
264  T* iter2 = &this->data[0] + c2;
265  int pixels = this->get_pixel_amount();
266  for (int i = 0; i < pixels; ++i, iter1 += this->c, iter2 += this->c)
267  std::swap(*iter1, *iter2);
268 }

270 template <typename T>
271 void
272 Image<T>::copy_channel (int src, int dest)
273 {
274  if (!this->valid() || src == dest)
275  return;
276 
277  if (dest < 0)
278  {
279  dest = this->channels();
280  this->add_channels(1);
281  }
282 
283  T const* src_iter = &this->data[0] + src;
284  T* dst_iter = &this->data[0] + dest;
285  int pixels = this->get_pixel_amount();
286  for (int i = 0; i < pixels; ++i, src_iter += this->c, dst_iter += this->c)
287  *dst_iter = *src_iter;
288 }

290 template <typename T>
291 void
292 Image<T>::delete_channel (int chan)
293 {
294  if (chan < 0 || chan >= this->channels())
295  return;
296 
297  typename std::vector<T>::iterator src_iter = this->data.begin();
298  typename std::vector<T>::iterator dst_iter = this->data.begin();
299  for (int i = 0; src_iter != this->data.end(); ++i)
300  {
301  if (i % this->c == chan)
302  src_iter++;
303  else
304  *(dst_iter++) = *(src_iter++);
305  }
306  this->resize(this->width(), this->height(), this->channels() - 1);
307 }

411 template <typename T>
412 T
413 Image<T>::linear_at (float x, float y, int channel) const
414 {
415  x = std::max(0.0f, std::min(static_cast<float>(this->w - 1), x));
416  y = std::max(0.0f, std::min(static_cast<float>(this->h - 1), y));
417 
418  int const floor_x = static_cast<int>(x);
419  int const floor_y = static_cast<int>(y);
420  int const floor_xp1 = std::min(floor_x + 1, this->w - 1);
421  int const floor_yp1 = std::min(floor_y + 1, this->h - 1);
422 
423  float const w1 = x - static_cast<float>(floor_x);
424  float const w0 = 1.0f - w1;
425  float const w3 = y - static_cast<float>(floor_y);
426  float const w2 = 1.0f - w3;
427 
428  int const rowstride = this->w * this->c;
429  int const row1 = floor_y * rowstride;
430  int const row2 = floor_yp1 * rowstride;
431  int const col1 = floor_x * this->c;
432  const int col2 = floor_xp1 * this->c;
433 
434  return math::interpolate<T>
435  (this->at(row1 + col1 + channel), this->at(row1 + col2 + channel),
436  this->at(row2 + col1 + channel), this->at(row2 + col2 + channel),
437  w0 * w2, w1 * w2, w0 * w3, w1 * w3);
438 }

440 template <typename T>
441 void
442 Image<T>::linear_at (float x, float y, T* px) const
443 {
444  x = std::max(0.0f, std::min(static_cast<float>(this->w - 1), x));
445  y = std::max(0.0f, std::min(static_cast<float>(this->h - 1), y));
446 
447  int const floor_x = static_cast<int>(x);
448  int const floor_y = static_cast<int>(y);
449  int const floor_xp1 = std::min(floor_x + 1, this->w - 1);
450  int const floor_yp1 = std::min(floor_y + 1, this->h - 1);
451 
452  float const w1 = x - static_cast<float>(floor_x);
453  float const w0 = 1.0f - w1;
454  float const w3 = y - static_cast<float>(floor_y);
455  float const w2 = 1.0f - w3;
456 
457  int const rowstride = this->w * this->c;
458  int const row1 = floor_y * rowstride;
459  int const row2 = floor_yp1 * rowstride;
460  int const col1 = floor_x * this->c;
461  int const col2 = floor_xp1 * this->c;
462 
463  /* Copy interpolated channel values to output buffer. */
464  for (int cc = 0; cc < this->c; ++cc)
465  {
466  px[cc] = math::interpolate<T>
467  (this->at(row1 + col1 + cc), this->at(row1 + col2 + cc),
468  this->at(row2 + col1 + cc), this->at(row2 + col2 + cc),
469  w0 * w2, w1 * w2, w0 * w3, w1 * w3);
470  }
471 }
