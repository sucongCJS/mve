275 inline ImageType
276 ImageBase::get_type_for_string (std::string const& type_string)
277 {
278  if (type_string == "sint8")
279  return IMAGE_TYPE_SINT8;
280  else if (type_string == "sint16")
281  return IMAGE_TYPE_SINT16;
282  else if (type_string == "sint32")
283  return IMAGE_TYPE_SINT32;
284  else if (type_string == "sint64")
285  return IMAGE_TYPE_SINT64;
286  else if (type_string == "uint8")
287  return IMAGE_TYPE_UINT8;
288  else if (type_string == "uint16")
289  return IMAGE_TYPE_UINT16;
290  else if (type_string == "uint32")
291  return IMAGE_TYPE_UINT32;
292  else if (type_string == "uint64")
293  return IMAGE_TYPE_UINT64;
294  else if (type_string == "float")
295  return IMAGE_TYPE_FLOAT;
296  else if (type_string == "double")
297  return IMAGE_TYPE_DOUBLE;
298 
299  return IMAGE_TYPE_UNKNOWN;
300 }
