129 template <typename T, typename FUNCTOR>
130 void
131 color_convert (typename Image<T>::Ptr image, FUNCTOR& converter)
132 {
133  int const channels = image->channels();
134  if (channels != 3)
135  throw std::invalid_argument("Only 3-channel images supported");
136 
137  for (T* ptr = image->begin(); ptr != image->end(); ptr += channels)
138  converter(ptr);
139 }

189 template <typename T>
190 void
191 color_xyy_to_xyz (T* v)
192 {
193  if (v[1] == T(0))
194  {
195  v[0] = T(0);
196  v[1] = T(0);
197  v[2] = T(0);
198  }
199  else
200  {
201  T const ratio = v[2] / v[1];
202  T out[3];
203  out[0] = v[0] * ratio;
204  out[1] = v[2];
205  out[2] = (T(1) - v[0] - v[1]) * ratio;
206  std::copy(out, out + 3, v);
207  }
208 }

235 color_xyz_to_xyy (T* v)
236 {

238  if (sum == T(0))

248  out[1] = v[1] / sum;

303 color_ycbcr_to_rgb (T* v)
304 {
305  v[1] = v[1] - T(0.5);
306  v[2] = v[2] - T(0.5);

308  T out[3];
309  out[0] = v[0] * T(1) + v[1] * T(0) + v[2] * T(1.402);
310  out[1] = v[0] * T(1) + v[1] * T(-0.34414) + v[2] * T(-0.71414);
311  out[2] = v[0] * T(1) + v[1] * T(1.772) + v[2] * T(0);
