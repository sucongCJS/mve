57 template <typename T>
58 void
59 draw_line (Image<T>& image, int x0, int y0, int x1, int y1, T const* color)
60 {
61  /* http://en.wikipedia.org/wiki/Bresenham%27s_line_algorithm */
62  int const chans = image.channels();
63  int const row_stride = image.width() * chans;
64 
65  int const dx = std::abs(x1 - x0);
66  int const dy = std::abs(y1 - y0) ;
67  int const sx = x0 < x1 ? 1 : -1;
68  int const sy = y0 < y1 ? 1 : -1;
69  int err = dx - dy;
70 
71  T* ptr = &image.at(x0, y0, 0);
72  while (true)
73  {
74  std::copy(color, color + chans, ptr);
75  if (x0 == x1 && y0 == y1)
76  break;
77  int const e2 = 2 * err;
78  if (e2 > -dy)
79  {
80  err -= dy;
81  x0 += sx;
82  ptr += sx * chans;
83  }
84  if (e2 < dx)
85  {
86  err += dx;
87  y0 += sy;
88  ptr += sy * row_stride;
89  }
90  }
91 }

94 template <typename T>
95 void
96 draw_circle (Image<T>& image, int x, int y, int radius, T const* color)
97 {
98  /* http://en.wikipedia.org/wiki/Bresenham_circle */
99  int const chans = image.channels();
100  int const row_stride = image.width() * chans;
101  T* const ptr = &image.at(x, y, 0);
102  std::copy(color, color + chans, ptr - radius * row_stride);
103  std::copy(color, color + chans, ptr - radius * chans);
104  std::copy(color, color + chans, ptr + radius * chans);
105  std::copy(color, color + chans, ptr + radius * row_stride);
106 
107  int f = 1 - radius;
108  int ddf_x = 1;
109  int ddf_y = -2 * radius;
110  int xi = 0;
111  int yi = radius;
112  while (xi < yi)
113  {
114  if (f >= 0)
115  {
116  yi--;
117  ddf_y += 2;
118  f += ddf_y;
119  }
120  xi++;
121  ddf_x += 2;
122  f += ddf_x;
123 
124  std::copy(color, color + chans, ptr + xi * chans + yi * row_stride);
125  std::copy(color, color + chans, ptr - xi * chans + yi * row_stride);
126  std::copy(color, color + chans, ptr + xi * chans - yi * row_stride);
127  std::copy(color, color + chans, ptr - xi * chans - yi * row_stride);
128  std::copy(color, color + chans, ptr + xi * row_stride + yi * chans);
129  std::copy(color, color + chans, ptr - xi * row_stride + yi * chans);
130  std::copy(color, color + chans, ptr + xi * row_stride - yi * chans);
131  std::copy(color, color + chans, ptr - xi * row_stride - yi * chans);
132  }
133 }

136 void
137 draw_rectangle (Image<T>& image, int x1, int y1, int x2, int y2, T const* color)
138 {
139  if (x1 > x2)
140  std::swap(x1, x2);
141  if (y1 > y2)
142  std::swap(y1, y2);
143  x1 = std::max(0, x1);
144  x2 = std::max(0, x2);
145  x1 = std::min(image.width() - 1, x1);
146  x2 = std::min(image.width() - 1, x2);
147  y1 = std::max(0, y1);
148  y2 = std::max(0, y2);
149  y1 = std::min(image.height() - 1, y1);
150  y2 = std::min(image.height() - 1, y2);
151 
152  int const row_stride = image.width() * image.channels();
153  T* ptr = image.begin();
154  for (int y = y1; y <= y2; ++y)
155  for (int x = x1; x <= x2; ++x)
156  std::copy(color, color + image.channels(),
157  ptr + x * image.channels() + y * row_stride);
158 }
