18 void MeshInfo::initialize (TriangleMesh::ConstPtr mesh)
19 {
20  TriangleMesh::VertexList const& verts = mesh->get_vertices();
21  TriangleMesh::FaceList const& faces = mesh->get_faces();
22  std::size_t face_amount = faces.size() / 3;
23 
24  this->vertex_info.clear();
25  this->vertex_info.resize(verts.size());
26 
27  /* Add faces to their three vertices. */
28  for (std::size_t i = 0, i3 = 0; i < face_amount; ++i)
29  for (std::size_t j = 0; j < 3; ++j, ++i3)
30  this->vertex_info[faces[i3]].faces.push_back(i);
31 
32  /* Classify each vertex and compute adjacenty info. */
33  for (std::size_t i = 0; i < this->vertex_info.size(); ++i)
34  this->update_vertex(*mesh, i);
35 }

54 void MeshInfo::update_vertex (TriangleMesh const& mesh, std::size_t vertex_id)
55 {
56  TriangleMesh::FaceList const& faces = mesh.get_faces();
57  VertexInfo& vinfo = this->vertex_info[vertex_id];
58 
59  /* Build new, temporary adjacent faces representation for ordering. */
60  AdjacentFaceList adj_temp;
61  for (std::size_t i = 0; i < vinfo.faces.size(); ++i)
62  {
63  std::size_t face_off = vinfo.faces[i] * 3;
64  for (std::size_t j = 0; j < 3; ++j)
65  if (faces[face_off + j] == vertex_id)
66  {
67  adj_temp.push_back(AdjacentFace());
68  adj_temp.back().face_id = vinfo.faces[i];
69  adj_temp.back().first = faces[face_off + (j + 1) % 3];
70  adj_temp.back().second = faces[face_off + (j + 2) % 3];
71  break;
72  }
73  }
74 
75  /* If there are no adjacent faces, the vertex is unreferenced. */
76  if (adj_temp.empty())
77  {
78  vinfo = VertexInfo();
79  vinfo.vclass = VERTEX_CLASS_UNREF;
80  return;
81  }
82 
83  /* Sort adjacent faces by chaining them. */
84  AdjacentFaceList adj_sorted;
85  adj_sorted.push_back(adj_temp.front());
86  adj_temp.pop_front();
87  while (!adj_temp.empty())
88  {
89  std::size_t const front_id = adj_sorted.front().first;
90  std::size_t const back_id = adj_sorted.back().second;
91 
92  /* Find a faces that fits the back or front of sorted list. */
93  bool found_face = false;
94  for (AdjacentFaceList::iterator iter = adj_temp.begin();
95  iter != adj_temp.end(); ++iter)
96  {
97  if (front_id == iter->second)
98  {
99  adj_sorted.push_front(*iter);
100  adj_temp.erase(iter);
101  found_face = true;
102  break;
103  }
104  if (back_id == iter->first)
105  {
106  adj_sorted.push_back(*iter);
107  adj_temp.erase(iter);
108  found_face = true;
109  break;
110  }
111  }
112 
113  /* If there is no next face, the vertex is complex. */
114  if (!found_face)
115  break;
116  }
117 
118  /* If the vertex is complex, add unsorted adjacency information. */
119  if (!adj_temp.empty())
120  {
121  /* Transfer remaining adjacent faces. */
122  adj_sorted.insert(adj_sorted.end(), adj_temp.begin(), adj_temp.end());
123  adj_temp.clear();
124 
125  /* Create unique list of all adjacent vertices. */
126  std::set<std::size_t> vset;
127  for (AdjacentFaceList::iterator iter = adj_sorted.begin();
128  iter != adj_sorted.end(); ++iter)
129  {
130  vset.insert(iter->first);
131  vset.insert(iter->second);
132  }
133  vinfo.verts.insert(vinfo.verts.end(), vset.begin(), vset.end());
134  vinfo.vclass = VERTEX_CLASS_COMPLEX;
135  return;
136  }
137 
138  /* If the vertex is not on the mesh boundary, the list is circular. */
139  if (adj_sorted.front().first == adj_sorted.back().second)
140  vinfo.vclass = VERTEX_CLASS_SIMPLE;
141  else
142  vinfo.vclass = VERTEX_CLASS_BORDER;
143 
144  /* Insert the face IDs in the adjacent faces list. */
145  vinfo.faces.clear();
146  for (AdjacentFaceList::iterator iter = adj_sorted.begin();
147  iter != adj_sorted.end(); ++iter)
148  vinfo.faces.push_back(iter->face_id);
149 
150  /* Insert vertex IDs in adjacent vertex list. */
151  for (AdjacentFaceList::const_iterator iter = adj_sorted.begin();
152  iter != adj_sorted.end(); ++iter)
153  vinfo.verts.push_back(iter->first);
154  if (vinfo.vclass == VERTEX_CLASS_BORDER)
155  vinfo.verts.push_back(adj_sorted.back().second);
156 }
