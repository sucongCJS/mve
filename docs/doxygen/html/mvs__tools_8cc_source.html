98 colAndExactDeriv(mve::ByteImage const& img, PixelCoords const& imgPos,
99  PixelCoords const& gradDir, Samples& color, Samples& deriv)
100 {
101  int const width = img.width();
102  int const height = img.height();
103  for (std::size_t i = 0; i < imgPos.size(); ++i)
104  {
105  int const left = std::floor(imgPos[i][0]);
106  int const top = std::floor(imgPos[i][1]);
107  float const x = imgPos[i][0] - left;
108  float const y = imgPos[i][1] - top;
109 
110  if (left < 0 || left > width-1 || top < 0 || top > height-1)
111  throw std::runtime_error("Image position out of bounds");
112 
113  /* data position pointer */
114  int p0 = (top * width + left) * 3;
115  int p1 = ((top+1) * width + left) * 3;
116 
117  /* bilinear interpolation to determine color value */
118  float x0, x1, x2, x3, x4, x5;
119  x0 = (1.f-x) * srgb2lin[img.at(p0 )] + x * srgb2lin[img.at(p0+3)];
120  x1 = (1.f-x) * srgb2lin[img.at(p0+1)] + x * srgb2lin[img.at(p0+4)];
121  x2 = (1.f-x) * srgb2lin[img.at(p0+2)] + x * srgb2lin[img.at(p0+5)];
122  x3 = (1.f-x) * srgb2lin[img.at(p1 )] + x * srgb2lin[img.at(p1+3)];
123  x4 = (1.f-x) * srgb2lin[img.at(p1+1)] + x * srgb2lin[img.at(p1+4)];
124  x5 = (1.f-x) * srgb2lin[img.at(p1+2)] + x * srgb2lin[img.at(p1+5)];
125 
126  color[i][0] = (1.f - y) * x0 + y * x3;
127  color[i][1] = (1.f - y) * x1 + y * x4;
128  color[i][2] = (1.f - y) * x2 + y * x5;
129 
130  /* derivative in direction gradDir */
131  float u = gradDir[i][0];
132  float v = gradDir[i][1];
133  for (int c = 0; c < 3; ++c)
134  {
135  deriv[i][c] =
136  u * (srgb2lin[img.at(p0+3)] - srgb2lin[img.at(p0)])
137  + v * (srgb2lin[img.at(p1)] - srgb2lin[img.at(p0)])
138  + (v * x + u * y) *
139  (srgb2lin[img.at(p0)] - srgb2lin[img.at(p0+3)]
140  - srgb2lin[img.at(p1)] + srgb2lin[img.at(p1+3)]);
141  ++p0;
142  ++p1;
143  }
144  }
145 }

168 void
169 getXYZColorAtPos(mve::ByteImage const& img, PixelCoords const& imgPos,
170  Samples* color)
171 {
172  int width = img.width();
173  int height = img.height();
174  PixelCoords::const_iterator citPos;
175  Samples::iterator itCol = color->begin();
176 
177  for (citPos = imgPos.begin(); citPos != imgPos.end(); ++citPos, ++itCol)
178  {
179  int const i = floor((*citPos)[0]);
180  int const j = floor((*citPos)[1]);
181  assert(i < width-1 && j < height-1);
182 
183  float const u = (*citPos)[0] - i;
184  float const v = (*citPos)[1] - j;
185  int const p0 = (j * width + i) * 3;
186  int const p1 = ((j+1) * width + i) * 3;
187  float x0, x1, x2, x3, x4, x5;
188  x0 = (1.f-u) * srgb2lin[img.at(p0 )] + u * srgb2lin[img.at(p0+3)];
189  x1 = (1.f-u) * srgb2lin[img.at(p0+1)] + u * srgb2lin[img.at(p0+4)];
190  x2 = (1.f-u) * srgb2lin[img.at(p0+2)] + u * srgb2lin[img.at(p0+5)];
191  x3 = (1.f-u) * srgb2lin[img.at(p1 )] + u * srgb2lin[img.at(p1+3)];
192  x4 = (1.f-u) * srgb2lin[img.at(p1+1)] + u * srgb2lin[img.at(p1+4)];
193  x5 = (1.f-u) * srgb2lin[img.at(p1+2)] + u * srgb2lin[img.at(p1+5)];
194 
195  (*itCol)[0] = (1.f - v) * x0 + v * x3;
196  (*itCol)[1] = (1.f - v) * x1 + v * x4;
197  (*itCol)[2] = (1.f - v) * x2 + v * x5;
198  }
199 }
