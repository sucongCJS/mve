45 inline float
46 parallax(math::Vec3f p, mvs::SingleView::Ptr v1, mvs::SingleView::Ptr v2)
47 {
48  math::Vec3f dir1 = (p - v1->camPos).normalized();
49  math::Vec3f dir2 = (p - v2->camPos).normalized();
50  float dp = std::max(std::min(dir1.dot(dir2), 1.f), -1.f);
51  float plx = std::acos(dp) * 180.f / pi;
52  return plx;
53 }
54 
55 inline float
56 parallaxToWeight(float p)
57 {
58  if (p < 0.f || p > 180.f) {
59  std::cerr << "ERROR: invalid parallax value." << std::endl;
60  return 0.f;
61  }
62  float sigma;
63  if (p <= 20.f)
64  sigma = 5.f;
65  else
66  sigma = 15.f;
67  float mean = 20.f;
68  return exp(- sqr(p - mean) / (2 * sigma * sigma));
69 }
