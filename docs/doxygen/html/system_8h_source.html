138  * Determine host byte-order.
139  * Windows: "All versions of windows run little-endian, period."

141  */
142 #if defined(_WIN32) || defined(__CYGWIN__)
143 # define HOST_BYTEORDER_LE
144 #endif
145 
146 #if defined(__APPLE__)
147 # if defined(__ppc__) || defined(__ppc64__)
148 # define HOST_BYTEORDER_BE
149 # elif defined(__i386__) || defined(__x86_64__)
150 # define HOST_BYTEORDER_LE
151 # endif
152 #endif
153 
154 #if defined(__linux__)
155 # include <endian.h> // part of glibc 2.9
156 # if __BYTE_ORDER == __LITTLE_ENDIAN
157 # define HOST_BYTEORDER_LE
158 # else
159 # define HOST_BYTEORDER_BE
160 # endif
161 #endif

163 #if defined(__FreeBSD__) || defined(__OpenBSD__) || defined(__NetBSD__)
164 # include <machine/endian.h>

167 # else
