369 inline Vector<T,3>
370 cross_product (Vector<T,3> const& v1, Vector<T,3> const& v2)
371 {
372  return Vector<T,3>(v1[1] * v2[2] - v1[2] * v2[1],
373  v1[2] * v2[0] - v1[0] * v2[2],
374  v1[0] * v2[1] - v1[1] * v2[0]);
